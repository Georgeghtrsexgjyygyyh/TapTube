// Phantom Agent Background Script
// Управляет системным VPN через Native Host (Xray Core)

let nativePort = null;
let isConnected = false;

// Connect to native host (xray.exe)
function connectNativeHost() {
  nativePort = chrome.runtime.connectNative('com.phantom.vpn.bridge');
  
  nativePort.onMessage.addListener((msg) => {
    if (msg.type === 'connected') {
      isConnected = true;
      updateIcon(true);
      broadcastToTabs({ type: 'VPN_STATUS', connected: true });
    } else if (msg.type === 'disconnected') {
      isConnected = false;
      updateIcon(false);
      broadcastToTabs({ type: 'VPN_STATUS', connected: false });
    } else if (msg.type === 'stats') {
      broadcastToTabs({ type: 'STATS_UPDATE', data: msg.data });
    }
  });
  
  nativePort.onDisconnect.addListener(() => {
    console.error('Native host disconnected:', chrome.runtime.lastError);
    isConnected = false;
    nativePort = null;
  });
}

// Handle commands from web page (via content script)
chrome.runtime.onMessageExternal.addListener((request, sender, sendResponse) => {
  if (request.action === 'connect') {
    if (!nativePort) connectNativeHost();
    nativePort.postMessage({ 
      cmd: 'start', 
      config: request.config 
    });
    sendResponse({ status: 'initiating' });
  } else if (request.action === 'disconnect') {
    if (nativePort) {
      nativePort.postMessage({ cmd: 'stop' });
    }
    sendResponse({ status: 'stopping' });
  }
  return true;
});

function updateIcon(active) {
  chrome.action.setIcon({
    path: active ? 'icon-active.png' : 'icon.png'
  });
}

function broadcastToTabs(message) {
  chrome.tabs.query({}, (tabs) => {
    tabs.forEach(tab => {
      chrome.tabs.sendMessage(tab.id, message).catch(() => {});
    });
  });
}

// Auto-connect on startup (optional)
chrome.runtime.onStartup.addListener(() => {
  console.log('Phantom Agent started');
});
        
// Bridge between Web Page and Extension
window.addEventListener('message', (event) => {
  if (event.source !== window) return;
  
  if (event.data.type === 'PHANTOM_COMMAND') {
    // Forward to background script
    chrome.runtime.sendMessage(chrome.runtime.id, event.data.payload, (response) => {
      window.postMessage({
        type: 'PHANTOM_RESPONSE',
        payload: response
      }, '*');
    });
  }
  
  if (event.data.type === 'PHANTOM_PING') {
    window.postMessage({
      type: 'PHANTOM_PONG',
      version: '2.0',
      timestamp: Date.now()
    }, '*');
  }
});

// Forward messages from background to page
chrome.runtime.onMessage.addListener((msg) => {
  window.postMessage(msg, '*');
});
        
#!/usr/bin/env python3
"""
Phantom Native Host
Запускает Xray Core и управляет им через stdin/stdout (Chrome Native Messaging protocol)
"""
import sys
import json
import struct
import subprocess
import threading
import os

def send_message(message):
    """Send message to Chrome"""
    encoded = json.dumps(message).encode('utf-8')
    sys.stdout.buffer.write(struct.pack('I', len(encoded)))
    sys.stdout.buffer.write(encoded)
    sys.stdout.flush()

def read_message():
    """Read message from Chrome"""
    raw_length = sys.stdin.buffer.read(4)
    if not raw_length:
        return None
    length = struct.unpack('I', raw_length)[0]
    message = sys.stdin.buffer.read(length).decode('utf-8')
    return json.loads(message)

def start_xray(config):
    """Start Xray process with given config"""
    # Generate config.json for Xray
    xray_config = {
        "log": {"loglevel": "warning"},
        "inbounds": [{
            "port": 10808,
            "protocol": "socks",
            "settings": {"auth": "noauth"}
        }],
        "outbounds": [{
            "protocol": "vless",
            "settings": {
                "vnext": [{
                    "address": config['server'].split(':')[0],
                    "port": int(config['server'].split(':')[1]),
                    "users": [{
                        "id": config['uuid'],
                        "flow": config['flow'],
                        "encryption": "none"
                    }]
                }]
            },
            "streamSettings": {
                "network": "tcp",
                "security": "reality",
                "realitySettings": {
                    "show": False,
                    "fingerprint": "chrome",
                    "serverName": config['dest'],
                    "publicKey": "YOUR_PUBLIC_KEY_HERE",
                    "shortId": "YOUR_SHORT_ID",
                    "spiderX": ""
                }
            }
        }]
    }
    
    # Save config
    with open('config.json', 'w') as f:
        json.dump(xray_config, f)
    
    # Start Xray
    process = subprocess.Popen(
        ['xray.exe', '-c', 'config.json'],
        stdout=subprocess.PIPE,
        stderr=subprocess.PIPE
    )
    return process

def main():
    xray_process = None
    
    while True:
        msg = read_message()
        if msg is None:
            break
            
        if msg.get('cmd') == 'start':
            try:
                xray_process = start_xray(msg['config'])
                send_message({"type": "connected", "status": "success"})
                
                # Monitor connection
                def monitor():
                    while True:
                        if xray_process.poll() is not None:
                            send_message({"type": "disconnected", "reason": "process_exit"})
                            break
                threading.Thread(target=monitor, daemon=True).start()
                
            except Exception as e:
                send_message({"type": "error", "message": str(e)})
                
        elif msg.get('cmd') == 'stop':
            if xray_process:
                xray_process.terminate()
                send_message({"type": "disconnected"})

if __name__ == '__main__':
    main()
